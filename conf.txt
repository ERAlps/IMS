FROM php:8.2-apache

# 安装连接 MySQL 所需的 pdo_mysql 扩展
RUN docker-php-ext-install pdo pdo_mysql

# 开启必要的 Apache 模块
RUN a2enmod rewrite headers

# 直接在全局配置中添加 MIME 类型支持，防止 .htaccess 不生效
RUN echo 'AddType application/javascript .tsx' >> /etc/apache2/mods-available/mime.conf && \
    echo 'AddType application/javascript .ts' >> /etc/apache2/mods-available/mime.conf

# 确保 /var/www/html 允许 .htaccess 覆盖
RUN sed -i '/<Directory \/var\/www\/>/,/<\/Directory>/ s/AllowOverride None/AllowOverride All/' /etc/apache2/apache2.conf

# 创建上传目录并设置权限
RUN mkdir -p /var/www/html/uploads && \
    chmod -R 777 /var/www/html/uploads && \
    chown -R www-data:www-data /var/www/html

# PHP 上传限制优化
RUN echo "upload_max_filesize = 100M\npost_max_size = 100M\nmemory_limit = 256M" > /usr/local/etc/php/conf.d/custom.ini

EXPOSE 80
